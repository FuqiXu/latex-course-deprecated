\section{Table with borders}

\begin{table}[h]

\centering
\label{t:borders}

\hline
1 & 2 & 3 \\ \hline
4 & 5 & 6 \\ \hline
7 & 8 & 9 \\ \hline
\end{tabular}
\end{table}
